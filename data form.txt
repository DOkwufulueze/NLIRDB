create table data_form(
matric_no char(10),
year_of_graduation char(10),
surname varchar(20),
other_names varchar(20),
name_of_school varchar(100),
gender char(6),
level varchar(12),
email_address varchar(30),
phone_no char(12),
date_of_birth date,
local_govt char(20),
state_of_origin char(20),
contact_address varchar(40),
permanent_address varchar(40),
aos char(50),
skill_set char(50),
sign char(120),
primary key(matric_no)
);
